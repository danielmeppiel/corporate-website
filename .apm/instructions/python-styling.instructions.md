    # Business logic here

    })

except ValidationError as e:
    # Log validation errors (sanitized)

        'error_type': type(e).__name__
    })

    # Log unexpected errors (no sensitive data)
    AuditLogger.log_event('system_error', {
        'error_type': type(e).__name__
    })
